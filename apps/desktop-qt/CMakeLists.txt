# QML aliases — root files first, then editors/* preserves the relative path so
# `import "editors"` resolves through the qmldir generated by qt_add_qml_module.
foreach(qml_file Main TabBar TabChip Sidebar Terminal BottomToolbar Theme
                 AIOverlay AIMessage SchemePicker
                 PaneView SplitContainer CommandPalette VoiceWidget
                 UpdateBanner LayoutDesigner SettingsPanel
                 EmptySlot VideoOpenDialog GridWorkspace FileTreeView
                 MovablePopup CheatsheetView AboutView)
    set_source_files_properties(
        ../../ui/qml/${qml_file}.qml
        PROPERTIES QT_RESOURCE_ALIAS ${qml_file}.qml
    )
endforeach()
foreach(qml_file EmojiPicker FavoriteEditor SnippetEditor CredentialEditor)
    set_source_files_properties(
        ../../ui/qml/editors/${qml_file}.qml
        PROPERTIES QT_RESOURCE_ALIAS editors/${qml_file}.qml
    )
endforeach()

# Theme.qml is a singleton (pragma Singleton).
set_source_files_properties(../../ui/qml/Theme.qml
    PROPERTIES QT_QML_SINGLETON_TYPE TRUE)

    AUTOMOC ON
    AUTORCC ON
    WIN32_EXECUTABLE ON
    MACOSX_BUNDLE ON
    OUTPUT_NAME "Dante CLI"
)

# Windows: copy Qt DLLs + plugins (platforms/, imageformats/, styles/) +
# the MSVC redistributable next to the .exe so the build folder is
# self-contained. The `--dir` flag is critical — without it windeployqt
# writes platforms/ relative to CWD (build/), not the .exe's directory
# (build/apps/desktop-qt/), and the installer ends up missing qwindows.dll
# which crashes the app on launch with no error dialog.
if(WIN32)
    add_custom_command(TARGET dante-cli POST_BUILD
        COMMAND Qt6::windeployqt
            --dir "$<TARGET_FILE_DIR:dante-cli>"
            --qmldir "${CMAKE_SOURCE_DIR}/ui/qml"
            --release
            --compiler-runtime
            "$<TARGET_FILE:dante-cli>"
        COMMENT "Running windeployqt with --dir + --compiler-runtime"
    )
endif()
